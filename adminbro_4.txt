-- HÃ m xá»­ lÃ½ chat
local function onChatted(message)
    if string.sub(message, 1, 1) == ";" or string.sub(message, 1, 1) == ":" then
        local args = {}
        for word in string.gmatch(message, "%S+") do
            table.insert(args, word)
        end
        
        local command = args[1]
        table.remove(args, 1)
        
        if commands[command] then
            commands[command].func(args)
        else
            game:GetService("ReplicatedStorage").DefaultChatSystemChatEvents.SayMessageRequest:FireServer("Lá»‡nh khÃ´ng tá»“n táº¡i! GÃµ ;help Ä‘á»ƒ xem danh sÃ¡ch lá»‡nh", "All")
        end
    end
end
